#include <stdio.h>
#include <winbgim.h>

#define MAX 50


void citesteSir(char s[MAX],int x, int y)
{
    strcpy(s,"");
    char s1[MAX];
    char t[2];
    char tasta;
    t[0]=tasta; t[1]='\0';
    strcat(s,t);
    strcpy(s1,s); strcat(s1,"_");
    setcolor(WHITE);
    outtextxy(x,y,s1);
    do {
        tasta=getch();
        if (tasta==8) // backspace
            if (strlen(s)>0)
                {
                    setcolor(BLACK);
                    strcpy(s1,s); strcat(s1,"_");
                    outtextxy(x,y,s1);
                    s[strlen(s)-1]='\0';
                    strcpy(s1,s); strcat(s1,"_");
                    outtextxy(x,y,s1);
                    setcolor(WHITE);
                    outtextxy(x,y,s1);
                }
            else
                Beep(1000,100);
        else
        {
            t[0]=tasta; t[1]='\0';
            strcat(s,t);
            strcpy(s1,s); strcat(s1,"_");
            setcolor(WHITE);
            outtextxy(x,y,s1);
        }
    } while (tasta!=13);
    t[0]=tasta; t[1]='\0';
    strcat(s,t);
    strcpy(s1,s); strcat(s1,"_");
    setcolor(BLACK); outtextxy(x,y,s1);
}

int main()
{
    char s[MAX];
    initwindow(800,600);
    citesteSir(s,100,100);
    if (!strcmp(s,"Gigi"))
        printf("Ai dat cuvantul corect.");
    else
        printf("Ai dat cuvantul gresit.");
    getch();
    closegraph();
}
